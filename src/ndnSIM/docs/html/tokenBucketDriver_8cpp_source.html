100 TBDriver::ScheduleNextToken( int bucket )
101 {
102  ns3::Ptr<ns3::Node> node= ns3::NodeContainer::GetGlobal().Get( ns3::Simulator::GetContext() );
103 
104  bool first;
105  double fillRate;
106 
107  if ( bucket == 0 ) {
108  first = m_first1;
109  fillRate = m_fillRate1;
110  } else if ( bucket == 1 ) {
111  first = m_first2;
112  fillRate = m_fillRate2;
113  } else {
114  first = m_first3;
115  fillRate = m_fillRate3;
116  }
117 
118  if ( first ) {
119  m_sendEvent = Simulator::Schedule( Seconds( 0.0 ), &TBDriver::UpdateBucket, this, bucket );
120  } else {
121  m_sendEvent = Simulator::Schedule( Seconds( 1.0 /fillRate ),
122  &TBDriver::UpdateBucket, this, bucket );
123  }

126 void
127 TBDriver::UpdateBucket( int bucket )
128 {
129  bool first;
130  double capacity;
131  nfd::fw::TokenBucket* sender;
132 
133  int node= ns3::NodeContainer::GetGlobal().Get( ns3::Simulator::GetContext() )->GetId();
134 
135  if ( m_connected == false && nfd::fw::CT.hasSender[node] == true ) {
136  m_connected = true;
137  }
138 
139  if ( bucket == 0 ) {
140  first = m_first1;
141  m_first1 = false;
142  capacity = m_capacity1;
143  sender = nfd::fw::CT.sender1[node];
144  } else if ( bucket == 1 ) {
145  first = m_first2;
146  m_first2 = false;
147  capacity = m_capacity2;
148  sender = nfd::fw::CT.sender2[node];
149  } else {
150  first = m_first3;
151  m_first3 = false;
152  capacity = m_capacity3;
153  sender = nfd::fw::CT.sender3[node];
154  }
155 
156  // Check to make sure tokens are not generated beyong specified capacity
157  if ( m_connected == true ) {
158  sender->m_capacity = capacity;
159  sender->addToken();
160  }
161 
162  ScheduleNextToken( bucket );
163 }
