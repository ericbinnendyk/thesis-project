75 static bool
76 isNextHopEligible(const Face& inFace, const Interest& interest,
77  const fib::NextHop& nexthop,
78  const shared_ptr<pit::Entry>& pitEntry,
79  bool wantUnused = false,
80  time::steady_clock::TimePoint now = time::steady_clock::TimePoint::min())
81 {
82  const Face& outFace = nexthop.getFace();
83 
84  // do not forward back to the same face, unless it is ad hoc
85  if (outFace.getId() == inFace.getId() && outFace.getLinkType() != ndn::nfd::LINK_TYPE_AD_HOC)
86  return false;
87 
88  // forwarding would violate scope
89  if (wouldViolateScope(inFace, interest, outFace))
90  return false;
91 
92  if (wantUnused) {
93  // nexthop must not have unexpired out-record
94  auto outRecord = pitEntry->getOutRecord(outFace);
95  if (outRecord != pitEntry->out_end() && outRecord->getExpiry() > now) {
96  return false;
97  }
98  }
99 
100  return true;
101 }

106 static fib::NextHopList::const_iterator
107 findEligibleNextHopWithEarliestOutRecord(const Face& inFace, const Interest& interest,
108  const fib::NextHopList& nexthops,
109  const shared_ptr<pit::Entry>& pitEntry)
110 {
111  auto found = nexthops.end();
112  auto earliestRenewed = time::steady_clock::TimePoint::max();
113 
114  for (auto it = nexthops.begin(); it != nexthops.end(); ++it) {
115  if (!isNextHopEligible(inFace, interest, *it, pitEntry))
116  continue;
117 
118  auto outRecord = pitEntry->getOutRecord(it->getFace());
119  BOOST_ASSERT(outRecord != pitEntry->out_end());
120  if (outRecord->getLastRenewed() < earliestRenewed) {
121  found = it;
122  earliestRenewed = outRecord->getLastRenewed();
123  }
124  }
125 
126  return found;
127 }

129 void
130 BestRouteStrategy2::afterReceiveInterest(const Face& inFace, const Interest& interest,
131  const shared_ptr<pit::Entry>& pitEntry)
132 {
133  RetxSuppressionResult suppression = m_retxSuppression.decidePerPitEntry(*pitEntry);
134  if (suppression == RetxSuppressionResult::SUPPRESS) {
135  NFD_LOG_DEBUG(interest << " from=" << inFace.getId()
136  << " suppressed");
137  return;
138  }
139 
140  const fib::Entry& fibEntry = this->lookupFib(*pitEntry);
141  const fib::NextHopList& nexthops = fibEntry.getNextHops();
142  auto it = nexthops.end();
143 
144 
145  int node= ns3::NodeContainer::GetGlobal().Get(ns3::Simulator::GetContext())->GetId();
146  if (suppression == RetxSuppressionResult::NEW) {
147  // forward to nexthop with lowest cost except downstream
148  it = std::find_if(nexthops.begin(), nexthops.end(), [&] (const auto& nexthop) {
149  return isNextHopEligible(inFace, interest, nexthop, pitEntry);
150  });
151 
152  if (it == nexthops.end()) {
153  NFD_LOG_DEBUG(interest << " from=" << inFace.getId() << " noNextHop");
154 
155  lp::NackHeader nackHeader;
156  nackHeader.setReason(lp::NackReason::NO_ROUTE);
157  this->sendNack(pitEntry, inFace, nackHeader);
158 
159  this->rejectPendingInterest(pitEntry);
160  return;
161  }
162 
163  Face& outFace = it->getFace();
164  this->sendInterest(pitEntry, outFace, interest);
165  NFD_LOG_DEBUG(interest << " from=" << inFace.getId()
166  << " newPitEntry-to=" << outFace.getId());
167  return;
168  }
169 
170  // find an unused upstream with lowest cost except downstream
171  it = std::find_if(nexthops.begin(), nexthops.end(), [&] (const auto& nexthop) {
172  return isNextHopEligible(inFace, interest, nexthop, pitEntry, true, time::steady_clock::now());
173  });
174 
175  if (it != nexthops.end()) {
176  Face& outFace = it->getFace();
177  this->sendInterest(pitEntry, outFace, interest);
178  NFD_LOG_DEBUG(interest << " from=" << inFace.getId()
179  << " retransmit-unused-to=" << outFace.getId());
180  return;
181  }
182 
183  // find an eligible upstream that is used earliest
184  it = findEligibleNextHopWithEarliestOutRecord(inFace, interest, nexthops, pitEntry);
185  if (it == nexthops.end()) {
186  NFD_LOG_DEBUG(interest << " from=" << inFace.getId() << " retransmitNoNextHop");
187  }
188  else {
189  Face& outFace = it->getFace();
190  this->sendInterest(pitEntry, outFace, interest);
191  NFD_LOG_DEBUG(interest << " from=" << inFace.getId()
192  << " retransmit-retry-to=" << outFace.getId());
193  }
194 }
