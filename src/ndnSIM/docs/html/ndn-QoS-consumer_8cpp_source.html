130 {

132 
133  m_sendEvent = Simulator::Schedule( Seconds( double( m_offset ) ), &QoSConsumer::SendPacket, this );

171 void
172 QoSConsumer::CheckRetxTimeout()
173 {
174  Time now = Simulator::Now();
175 
176  Time rto = m_rtt->RetransmitTimeout();
177 
178  //NS_LOG_DEBUG ( "Current RTO: " << rto.ToDouble ( Time::S ) << "s" );
179 
180  while ( !m_seqTimeouts.empty() ) {
181 
182  SeqTimeoutsContainer::index<i_timestamp>::type::iterator entry =
183  m_seqTimeouts.get<i_timestamp>().begin();
184 
185  // timeout expired?
186  if ( entry->time + rto <= now ) {
187  uint32_t seqNo = entry->seq;
188  m_seqTimeouts.get<i_timestamp>().erase( entry );
189  OnTimeout( seqNo );
190  } else {
191  break; // nothing else to do. All later packets need not be retransmitted
192  }
193  }
194 
195  m_retxEvent = Simulator::Schedule( m_retxTimer, &QoSConsumer::CheckRetxTimeout, this );
196 }

217 void
218 QoSConsumer::SendPacket()
219 {
220  // Set default size for payload interets
221  if ( m_subscription == 0 && m_virtualPayloadSize == 0 ) {
222  m_virtualPayloadSize = 4;
223  }
224 
225  if ( !m_active ) {
226  return;
227  }
228 
229  NS_LOG_FUNCTION_NOARGS();
230 
231  uint32_t seq = std::numeric_limits<uint32_t>::max();
232 
233  while ( m_retxSeqs.size() ) {
234 
235  seq = *m_retxSeqs.begin();
236  m_retxSeqs.erase( m_retxSeqs.begin() );
237  break;
238  }
239 
240  if ( m_seqMax != std::numeric_limits<uint32_t>::max() ) {
241 
242  if ( m_seq >= m_seqMax ) {
243  return;
244  }
245  }
246 
247  seq = m_seq++;
248  uint8_t payload[1] = {1};
249 
250  shared_ptr<Name> nameWithSequence = make_shared<Name>( m_interestName );
251 
252  shared_ptr<Interest> interest = make_shared<Interest>();
253  interest->setNonce( m_rand->GetValue( 0, std::numeric_limits<uint32_t>::max() ) );
254  interest->setSubscription( m_subscription );
255  nameWithSequence->appendSequenceNumber( seq );
256 
257  if ( m_subscription == 0 ) {
258  interest->setPayload( payload, m_virtualPayloadSize ); //add payload to interest
259  }
260 
261  interest->setName( *nameWithSequence );
262  time::milliseconds interestLifeTime( m_interestLifeTime.GetMilliSeconds() );
263  interest->setInterestLifetime( interestLifeTime );
264 
265  NS_LOG_INFO( "node( " << GetNode()->GetId() << " ) > sending Interest: " << interest->getName() /*m_interestName*/ << " with Payload = " << interest->getPayloadLength() << "bytes" );
266 
267  WillSendOutInterest( seq );
268 
269  m_transmittedInterests( interest, this, m_face );
270  m_appLink->onReceiveInterest( *interest );
271 
272  // Callback for sent payload interests
273  m_sentInterest( GetNode()->GetId(), interest );
274 
275  ScheduleNextPacket();
276 }

283 void
284 QoSConsumer::OnData( shared_ptr<const Data> data )
285 {
286 
287  if ( !m_active ) {
288  return;
289  }
290 
291  App::OnData( data ); // tracing inside
292 
293  NS_LOG_FUNCTION( this << data );
294 
295  // This could be a problem......
296  //uint32_t seq = data->getName().at( -1 ).toSequenceNumber();
297 
298  NS_LOG_INFO( "node( " << GetNode()->GetId() << " ) < Received DATA for " << data->getName() << " TIME: " << Simulator::Now() );
299 
300  // Callback for received subscription data
301  m_receivedData( GetNode()->GetId(), data );
302 
303  int hopCount = 0;
304  auto hopCountTag = data->getTag<lp::HopCountTag>();
305 
306  if ( hopCountTag != nullptr ) { // e.g., packet came from local node's cache
307  hopCount = *hopCountTag;
308  }
309 
310  NS_LOG_DEBUG( "Hop count: " << hopCount );
311 
312  // Enable trace file for Interests with sequence number ( subscription = 0 )
313  if ( m_subscription == 0 ) {
314 
315  // This could be a problem......
316  uint32_t seq = data->getName().at( -1 ).toSequenceNumber();
317  SeqTimeoutsContainer::iterator entry = m_seqLastDelay.find( seq );
318 
319  if ( entry != m_seqLastDelay.end() ) {
320  m_lastRetransmittedInterestDataDelay( this, seq, Simulator::Now() - entry->time, hopCount );
321  }
322 
323  entry = m_seqFullDelay.find( seq );
324 
325  if ( entry != m_seqFullDelay.end() ) {
326  m_firstInterestDataDelay( this, seq, Simulator::Now() - entry->time, m_seqRetxCounts[seq], hopCount );
327  }
328 
329  m_seqRetxCounts.erase( seq );
330  m_seqFullDelay.erase( seq );
331  m_seqLastDelay.erase( seq );
332  m_seqTimeouts.erase( seq );
333  m_retxSeqs.erase( seq );
334  m_rtt->AckSeq( SequenceNumber32( seq ) );
335  }
336 
337  //data = nullptr;
338  //data.reset();
339  //std::cout << "data received at APP in subscriber " << data->getName() << " count " << data.use_count() << std::endl;
340 }

343 QoSConsumer::OnTimeout( uint32_t sequenceNumber )
344 {
345  //NS_LOG_FUNCTION( sequenceNumber );

347  //m_rtt->RetransmitTimeout ().ToDouble ( Time::S ) << "s\n";
348 
349  m_rtt->IncreaseMultiplier(); // Double the next RTO

355 void
356 QoSConsumer::WillSendOutInterest( uint32_t sequenceNumber )
357 {
358  NS_LOG_DEBUG( "Trying to add " << sequenceNumber << " with " << Simulator::Now() << ". already "
359  << m_seqTimeouts.size() << " items" );
360  //m_seqTimeouts.insert( SeqTimeout( sequenceNumber, Simulator::Now() ) );

362  m_seqLastDelay.erase( sequenceNumber );
363  m_seqLastDelay.insert( SeqTimeout( sequenceNumber, Simulator::Now() ) );
364  m_seqRetxCounts[sequenceNumber]++;
365  m_rtt->SentSeq( SequenceNumber32( sequenceNumber ), 1 );
