86 void
87 Forwarder::onIncomingInterest(Face& inFace, const Interest& interest)
88 {
89  // receive Interest
90  NFD_LOG_DEBUG("onIncomingInterest face=" << inFace.getId() <<
91  " interest=" << interest.getName());
92  interest.setTag(make_shared<lp::IncomingFaceIdTag>(inFace.getId()));
93  ++m_counters.nInInterests;
94 
95  // /localhost scope control
96  bool isViolatingLocalhost = inFace.getScope() == ndn::nfd::FACE_SCOPE_NON_LOCAL &&
97  scope_prefix::LOCALHOST.isPrefixOf(interest.getName());
98  if (isViolatingLocalhost) {
99  NFD_LOG_DEBUG("onIncomingInterest face=" << inFace.getId() <<
100  " interest=" << interest.getName() << " violates /localhost");
101  // (drop)
102  return;
103  }
104 
105  // detect duplicate Nonce with Dead Nonce List
106  bool hasDuplicateNonceInDnl = m_deadNonceList.has(interest.getName(), interest.getNonce());
107  if (hasDuplicateNonceInDnl) {
108  // goto Interest loop pipeline
109  this->onInterestLoop(inFace, interest);
110  return;
111  }
112 
113  // strip forwarding hint if Interest has reached producer region
114  if (!interest.getForwardingHint().empty() &&
115  m_networkRegionTable.isInProducerRegion(interest.getForwardingHint())) {
116  NFD_LOG_DEBUG("onIncomingInterest face=" << inFace.getId() <<
117  " interest=" << interest.getName() << " reaching-producer-region");
118  const_cast<Interest&>(interest).setForwardingHint({});
119  }
120 
121  // PIT insert
122  shared_ptr<pit::Entry> pitEntry = m_pit.insert(interest).first;
123 
124  // detect duplicate Nonce in PIT entry
125  int dnw = fw::findDuplicateNonce(*pitEntry, interest.getNonce(), inFace);
126  bool hasDuplicateNonceInPit = dnw != fw::DUPLICATE_NONCE_NONE;
127  if (inFace.getLinkType() == ndn::nfd::LINK_TYPE_POINT_TO_POINT) {
128  // for p2p face: duplicate Nonce from same incoming face is not loop
129  hasDuplicateNonceInPit = hasDuplicateNonceInPit && !(dnw & fw::DUPLICATE_NONCE_IN_SAME);
130  }
131  if (hasDuplicateNonceInPit) {
132  // goto Interest loop pipeline
133  this->onInterestLoop(inFace, interest);
134  return;
135  }
136 
137  // is pending?
138  if (!pitEntry->hasInRecords()) {
139  if (m_csFromNdnSim == nullptr) {
140  m_cs.find(interest,
141  bind(&Forwarder::onContentStoreHit, this, std::ref(inFace), pitEntry, _1, _2),
142  bind(&Forwarder::onContentStoreMiss, this, std::ref(inFace), pitEntry, _1));
143  }
144  else {
145  shared_ptr<Data> match = m_csFromNdnSim->Lookup(interest.shared_from_this());
146  if (match != nullptr) {
147  this->onContentStoreHit(inFace, pitEntry, interest, *match);
148  }
149  else {
150  this->onContentStoreMiss(inFace, pitEntry, interest);
151  }
152  }
153  }
154  else {
155  this->onContentStoreMiss(inFace, pitEntry, interest);
156  }
157 }

187 void
188 Forwarder::onContentStoreMiss(const Face& inFace, const shared_ptr<pit::Entry>& pitEntry,
189  const Interest& interest)
190 {
191  NFD_LOG_DEBUG("onContentStoreMiss interest=" << interest.getName());
192  ++m_counters.nCsMisses;
193 
194  // insert in-record
195  pitEntry->insertOrUpdateInRecord(const_cast<Face&>(inFace), interest);
196 
197  // set PIT expiry timer to the time that the last PIT in-record expires
198  auto lastExpiring = std::max_element(pitEntry->in_begin(), pitEntry->in_end(), &compare_InRecord_expiry);
199  auto lastExpiryFromNow = lastExpiring->getExpiry() - time::steady_clock::now();
200  this->setExpiryTimer(pitEntry, time::duration_cast<time::milliseconds>(lastExpiryFromNow));
201 
202  // has NextHopFaceId?
203  shared_ptr<lp::NextHopFaceIdTag> nextHopTag = interest.getTag<lp::NextHopFaceIdTag>();
204  if (nextHopTag != nullptr) {
205  // chosen NextHop face exists?
206  Face* nextHopFace = m_faceTable.get(*nextHopTag);
207  if (nextHopFace != nullptr) {
208  NFD_LOG_DEBUG("onContentStoreMiss interest=" << interest.getName() << " nexthop-faceid=" << nextHopFace->getId());
209  // go to outgoing Interest pipeline
210  // scope control is unnecessary, because privileged app explicitly wants to forward
211  this->onOutgoingInterest(pitEntry, *nextHopFace, interest);
212  }
213  return;
214  }
215 
216  // dispatch to strategy: after incoming Interest
217  this->dispatchToStrategy(*pitEntry,
218  [&] (fw::Strategy& strategy) { strategy.afterReceiveInterest(inFace, interest, pitEntry); });
219 }

260 void
261 Forwarder::onInterestFinalize(const shared_ptr<pit::Entry>& pitEntry)
262 {
263  NFD_LOG_DEBUG("onInterestFinalize interest=" << pitEntry->getName() <<
264  (pitEntry->isSatisfied ? " satisfied" : " unsatisfied"));
265  if (!pitEntry->isSatisfied) {
266  beforeExpirePendingInterest(*pitEntry);
267  }
268  // Dead Nonce List insert if necessary
269  this->insertDeadNonceList(*pitEntry, 0);
270 
271  // Increment satisfied/unsatisfied Interests counter
272  if (pitEntry->isSatisfied) {
273  ++m_counters.nSatisfiedInterests;
274  }
275  else {
276  ++m_counters.nUnsatisfiedInterests;
277  }
278 
279  // PIT delete
280  scheduler::cancel(pitEntry->expiryTimer);
281  m_pit.erase(pitEntry.get());

284 void
285 Forwarder::onIncomingData(Face& inFace, const Data& data)
286 {
287  // receive Data
288  NFD_LOG_DEBUG("onIncomingData face=" << inFace.getId() << " data=" << data.getName());
289  data.setTag(make_shared<lp::IncomingFaceIdTag>(inFace.getId()));
290  ++m_counters.nInData;
291 
292  // /localhost scope control
293  bool isViolatingLocalhost = inFace.getScope() == ndn::nfd::FACE_SCOPE_NON_LOCAL &&
294  scope_prefix::LOCALHOST.isPrefixOf(data.getName());
295  if (isViolatingLocalhost) {
296  NFD_LOG_DEBUG("onIncomingData face=" << inFace.getId() <<
297  " data=" << data.getName() << " violates /localhost");
298  // (drop)
299  return;
300  }
301 
302  // PIT match
303  pit::DataMatchResult pitMatches = m_pit.findAllDataMatches(data);
304  if (pitMatches.size() == 0) {
305  // goto Data unsolicited pipeline
306  this->onDataUnsolicited(inFace, data);
307  return;
308  }
309 
310  shared_ptr<Data> dataCopyWithoutTag = make_shared<Data>(data);
311  dataCopyWithoutTag->removeTag<lp::HopCountTag>();
312 
313  // CS insert
314  if (m_csFromNdnSim == nullptr)
315  m_cs.insert(*dataCopyWithoutTag);
316  else
317  m_csFromNdnSim->Add(dataCopyWithoutTag);
318 
319  // when only one PIT entry is matched, trigger strategy: after receive Data
320  if (pitMatches.size() == 1) {
321  auto& pitEntry = pitMatches.front();
322 
323  NFD_LOG_DEBUG("onIncomingData matching=" << pitEntry->getName());
324 
325  // set PIT expiry timer to now
326  this->setExpiryTimer(pitEntry, 0_ms);
327 
328  beforeSatisfyInterest(*pitEntry, inFace, data);
329  // trigger strategy: after receive Data
330  this->dispatchToStrategy(*pitEntry,
331  [&] (fw::Strategy& strategy) { strategy.afterReceiveData(pitEntry, inFace, data); });
332 
333  // mark PIT satisfied
334  pitEntry->isSatisfied = true;
335  pitEntry->dataFreshnessPeriod = data.getFreshnessPeriod();
336 
337  // Dead Nonce List insert if necessary (for out-record of inFace)
338  this->insertDeadNonceList(*pitEntry, &inFace);
339 
340  // delete PIT entry's out-record
341  pitEntry->deleteOutRecord(inFace);
342  }
343  // when more than one PIT entry is matched, trigger strategy: before satisfy Interest,
344  // and send Data to all matched out faces
345  else {
346  std::set<Face*> pendingDownstreams;
347  auto now = time::steady_clock::now();
348 
349  for (const shared_ptr<pit::Entry>& pitEntry : pitMatches) {
350  NFD_LOG_DEBUG("onIncomingData matching=" << pitEntry->getName());
351 
352  // remember pending downstreams
353  for (const pit::InRecord& inRecord : pitEntry->getInRecords()) {
354  if (inRecord.getExpiry() > now) {
355  pendingDownstreams.insert(&inRecord.getFace());
356  }
357  }
358 
359  // set PIT expiry timer to now
360  this->setExpiryTimer(pitEntry, 0_ms);
361 
362  // invoke PIT satisfy callback
363  beforeSatisfyInterest(*pitEntry, inFace, data);
364  this->dispatchToStrategy(*pitEntry,
365  [&] (fw::Strategy& strategy) { strategy.beforeSatisfyInterest(pitEntry, inFace, data); });
366 
367  // mark PIT satisfied
368  pitEntry->isSatisfied = true;
369  pitEntry->dataFreshnessPeriod = data.getFreshnessPeriod();
370 
371  // Dead Nonce List insert if necessary (for out-record of inFace)
372  this->insertDeadNonceList(*pitEntry, &inFace);
373 
374  // clear PIT entry's in and out records
375  pitEntry->clearInRecords();
376  pitEntry->deleteOutRecord(inFace);
377  }
378 
379  // foreach pending downstream
380  for (Face* pendingDownstream : pendingDownstreams) {
381  if (pendingDownstream->getId() == inFace.getId() &&
382  pendingDownstream->getLinkType() != ndn::nfd::LINK_TYPE_AD_HOC) {
383  continue;
384  }
385  // goto outgoing Data pipeline
386  this->onOutgoingData(data, *pendingDownstream);
387  }
388  }
389 }

435 void
436 Forwarder::onIncomingNack(Face& inFace, const lp::Nack& nack)
437 {
438  // receive Nack
439  nack.setTag(make_shared<lp::IncomingFaceIdTag>(inFace.getId()));
440  ++m_counters.nInNacks;
441 
442  // if multi-access or ad hoc face, drop
443  if (inFace.getLinkType() != ndn::nfd::LINK_TYPE_POINT_TO_POINT) {
444  NFD_LOG_DEBUG("onIncomingNack face=" << inFace.getId() <<
445  " nack=" << nack.getInterest().getName() <<
446  "~" << nack.getReason() << " face-is-multi-access");
447  return;
448  }
449 
450  // PIT match
451  shared_ptr<pit::Entry> pitEntry = m_pit.find(nack.getInterest());
452  // if no PIT entry found, drop
453  if (pitEntry == nullptr) {
454  NFD_LOG_DEBUG("onIncomingNack face=" << inFace.getId() <<
455  " nack=" << nack.getInterest().getName() <<
456  "~" << nack.getReason() << " no-PIT-entry");
457  return;
458  }
459 
460  // has out-record?
461  pit::OutRecordCollection::iterator outRecord = pitEntry->getOutRecord(inFace);
462  // if no out-record found, drop
463  if (outRecord == pitEntry->out_end()) {
464  NFD_LOG_DEBUG("onIncomingNack face=" << inFace.getId() <<
465  " nack=" << nack.getInterest().getName() <<
466  "~" << nack.getReason() << " no-out-record");
467  return;
468  }
469 
470  // if out-record has different Nonce, drop
471  if (nack.getInterest().getNonce() != outRecord->getLastNonce()) {
472  NFD_LOG_DEBUG("onIncomingNack face=" << inFace.getId() <<
473  " nack=" << nack.getInterest().getName() <<
474  "~" << nack.getReason() << " wrong-Nonce " <<
475  nack.getInterest().getNonce() << "!=" << outRecord->getLastNonce());
476  return;
477  }
478 
479  NFD_LOG_DEBUG("onIncomingNack face=" << inFace.getId() <<
480  " nack=" << nack.getInterest().getName() <<
481  "~" << nack.getReason() << " OK");
482 
483  // record Nack on out-record
484  outRecord->setIncomingNack(nack);
485 
486  // set PIT expiry timer to now when all out-record receive Nack
487  if (!fw::hasPendingOutRecords(*pitEntry)) {
488  this->setExpiryTimer(pitEntry, 0_ms);
489  }
490 
491  // trigger strategy: after receive NACK
492  this->dispatchToStrategy(*pitEntry,
493  [&] (fw::Strategy& strategy) { strategy.afterReceiveNack(inFace, nack, pitEntry); });
494 }

506

508  pit::InRecordCollection::iterator inRecord = pitEntry->getInRecord(outFace);

512  NFD_LOG_DEBUG("onOutgoingNack face=" << outFace.getId() <<
513  " nack=" << pitEntry->getInterest().getName() <<
514  "~" << nack.getReason() << " no-in-record");
515  return;
516  }
517 
518  // if multi-access or ad hoc face, drop
519  if (outFace.getLinkType() != ndn::nfd::LINK_TYPE_POINT_TO_POINT) {
520  NFD_LOG_DEBUG("onOutgoingNack face=" << outFace.getId() <<
521  " nack=" << pitEntry->getInterest().getName() <<
522  "~" << nack.getReason() << " face-is-multi-access");
523  return;
524  }
525 
526  NFD_LOG_DEBUG("onOutgoingNack face=" << outFace.getId() <<
527  " nack=" << pitEntry->getInterest().getName() <<
528  "~" << nack.getReason() << " OK");
529 
530  // create Nack packet with the Interest from in-record
531  lp::Nack nackPkt(inRecord->getInterest());
532  nackPkt.setHeader(nack);
533 
534  // erase in-record
535  pitEntry->deleteInRecord(outFace);
536 
537  // send Nack on face
538  const_cast<Face&>(outFace).sendNack(nackPkt);
539  ++m_counters.nOutNacks;

559 void
560 Forwarder::insertDeadNonceList(pit::Entry& pitEntry, Face* upstream)
561 {
562  // need Dead Nonce List insert?
563  bool needDnl = true;
564  if (pitEntry.isSatisfied) {
565  BOOST_ASSERT(pitEntry.dataFreshnessPeriod >= 0_ms);
566  needDnl = static_cast<bool>(pitEntry.getInterest().getMustBeFresh()) &&
567  pitEntry.dataFreshnessPeriod < m_deadNonceList.getLifetime();
568  }
569 
570  if (!needDnl) {
571  return;
572  }
573 
574  // Dead Nonce List insert
575  if (upstream == nullptr) {
576  // insert all outgoing Nonces
577  const auto& outRecords = pitEntry.getOutRecords();
578  std::for_each(outRecords.begin(), outRecords.end(), [&] (const auto& outRecord) {
579  m_deadNonceList.add(pitEntry.getName(), outRecord.getLastNonce());
580  });
581  }
582  else {
583  // insert outgoing Nonce of a specific face
584  auto outRecord = pitEntry.getOutRecord(*upstream);
585  if (outRecord != pitEntry.getOutRecords().end()) {
586  m_deadNonceList.add(pitEntry.getName(), outRecord->getLastNonce());
587  }
588  }
589 }
