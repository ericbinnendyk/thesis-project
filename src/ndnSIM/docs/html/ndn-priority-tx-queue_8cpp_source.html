92 NdnPriorityTxQueue::DoEnqueue( QueueItem item, uint32_t pr_level )
93 {
94  QosQueue *queue;
95 
96  if( pr_level >= 1 && pr_level <= 20 ) {
97  queue = &m_highPriorityQueue;

99  queue = &m_mediumPriorityQueue;

103  return false;
104  }
105 
106  if( queue->Enqueue( item ) ) {

109  } else {
