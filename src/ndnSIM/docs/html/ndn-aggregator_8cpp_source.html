156 void
157 Aggregator::ScheduleAggPackets()
158 {
159  //NS_LOG_FUNCTION_NOARGS();
160 
161  if (m_firstTime == true) {
162  m_firstTime = false;
163  //Schedule next interest with aggreagated payload
164  //Set the offset for AMI/PMU traffic to milliseconds
165  m_txEvent = Simulator::Schedule(Seconds( double(m_offset)/1000 ), &Aggregator::ScheduleAggPackets, this);
166  }
167  else {
168  //Schedule next interest with aggregated payload
169  if (!m_txEvent.IsRunning())
170  m_txEvent = Simulator::Schedule(m_frequency, &Aggregator::SendAggInterest, this);
171  }

174 void
175 Aggregator::SendAggInterest()
176 {
177  //If no payloaded interest received, do not send an interest with zero payload
178  if (m_totalpayload > 0)
179  {
180 
181  if (!m_active)
182  return;
183 
184  uint32_t seq = std::numeric_limits<uint32_t>::max();
185 
186  while (m_retxSeqs.size()) {
187  seq = *m_retxSeqs.begin();
188  m_retxSeqs.erase(m_retxSeqs.begin());
189  break;
190  }
191 
192  if (m_seqMax != std::numeric_limits<uint32_t>::max()) {
193  if (m_seq >= m_seqMax) {
194  return;
195  }
196  }
197 
198  seq = m_seq++;
199 
200  uint8_t payload[1] = {1};
201 
202 
203  //Append source aggregation node ID to the aggregated prefix
204  std::stringstream ss;
205  ss << m_upstream_prefix;
206  std::string dst_com_prefix = ss.str() + "/agg" + std::to_string(GetNode()->GetId());
207 
208  //
209  shared_ptr<Name> nameWithSequence = make_shared<Name>(dst_com_prefix);
210  nameWithSequence->appendSequenceNumber(seq);
211  //
212 
213  shared_ptr<Interest> interest = make_shared<Interest>();
214  interest->setNonce(m_rand->GetValue(0, std::numeric_limits<uint32_t>::max()));
215  interest->setName(*nameWithSequence);
216  interest->setSubscription(0); //not a subscription interest
217  time::milliseconds interestLifeTime(m_interestLifeTime.GetMilliSeconds());
218  interest->setInterestLifetime(interestLifeTime);
219  interest->setPayload(payload, m_totalpayload); //concatenated payload
220 
221  NS_LOG_INFO("node(" << GetNode()->GetId() << ") > forwarding Interest: " << interest->getName() << " with aggregated Payload = " << m_totalpayload << " TIME: " << Simulator::Now());
222 
223  WillSendOutInterest(seq);
224 
225  m_transmittedInterests(interest, this, m_face);
226  m_appLink->onReceiveInterest(*interest);
227 
228  //reset the total payload size to start accumulating subsequent ones
229  m_totalpayload = 0;
230 
231  //Call back for sent aggregated interests
232  m_sentInterest(GetNode()->GetId(), interest);
233  }
234 
235  //Shedule next aggregated payload interest
236  ScheduleAggPackets();
237 }

239 void
240 Aggregator::WillSendOutInterest(uint32_t sequenceNumber)
241 {
242  NS_LOG_DEBUG("Trying to add " << sequenceNumber << " with " << Simulator::Now() << ". already "
243  << m_seqTimeouts.size() << " items");
244 
245  m_seqTimeouts.insert(SeqTimeout(sequenceNumber, Simulator::Now()));
246  m_seqFullDelay.insert(SeqTimeout(sequenceNumber, Simulator::Now()));
247 
248  m_seqLastDelay.erase(sequenceNumber);
249  m_seqLastDelay.insert(SeqTimeout(sequenceNumber, Simulator::Now()));
250 
251  m_seqRetxCounts[sequenceNumber]++;
252 
253  m_rtt->SentSeq(SequenceNumber32(sequenceNumber), 1);
254 }

276 void
277 Aggregator::CheckRetxTimeout()
278 {
279  Time now = Simulator::Now();
280 
281  Time rto = m_rtt->RetransmitTimeout();
282  // NS_LOG_DEBUG ("Current RTO: " << rto.ToDouble (Time::S) << "s");
283 
284  while (!m_seqTimeouts.empty()) {
285  SeqTimeoutsContainer::index<i_timestamp>::type::iterator entry =
286  m_seqTimeouts.get<i_timestamp>().begin();
287 
288  if (entry->time + rto <= now) // timeout expired?
289  {
290  uint32_t seqNo = entry->seq;
291  m_seqTimeouts.get<i_timestamp>().erase(entry);
292  OnTimeout(seqNo);
293  }
294  else
295  break; // nothing else to do. All later packets need not be retransmitted
296  }
297 
298  m_retxEvent = Simulator::Schedule(m_retxTimer, &Aggregator::CheckRetxTimeout, this);
299 }

301 void
302 Aggregator::OnTimeout(uint32_t sequenceNumber)
303 {
304  m_rtt->IncreaseMultiplier(); // Double the next RTO
305  m_rtt->SentSeq(SequenceNumber32(sequenceNumber),
306  1); // make sure to disable RTT calculation for this sample
307  m_retxSeqs.insert(sequenceNumber);
308 
309  ScheduleAggPackets();
310 }
