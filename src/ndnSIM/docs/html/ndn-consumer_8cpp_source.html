111 void
112 Consumer::CheckRetxTimeout()
113 {
114  Time now = Simulator::Now();
115 
116  Time rto = m_rtt->RetransmitTimeout();
117  // NS_LOG_DEBUG ("Current RTO: " << rto.ToDouble (Time::S) << "s");
118 
119  while (!m_seqTimeouts.empty()) {
120  SeqTimeoutsContainer::index<i_timestamp>::type::iterator entry =
121  m_seqTimeouts.get<i_timestamp>().begin();
122  if (entry->time + rto <= now) // timeout expired?
123  {
124  uint32_t seqNo = entry->seq;
125  m_seqTimeouts.get<i_timestamp>().erase(entry);
126  OnTimeout(seqNo);
127  }
128  else
129  break; // nothing else to do. All later packets need not be retransmitted
130  }
131 
132  m_retxEvent = Simulator::Schedule(m_retxTimer, &Consumer::CheckRetxTimeout, this);
133 }
