140 void
141 SpontaneousProducer::SendTimeout(){
142 
143  double send_delay = 0.0; //(double)((Simulator::Now().GetNanoSeconds())/1000000000);
144 
145  //Do not send initial data before scheduling with the input frequency
146  if(m_firstTime) {
147  m_firstTime = false;
148  } else {
149  //Only send data when there is a subscription (1-soft or 2-hard)
150  if (m_subscription == 1 || m_subscription == 2) {
151  //Send multiple chunks of 1Kbyte (1024bytes) data to physical node
152  for (int i=0; i<(int)m_subDataSize; i++) {
153  //SendData(m_prefix);
154  Simulator::Schedule(Seconds(send_delay), &SpontaneousProducer::SendData, this, m_prefix);
155  send_delay = send_delay + 0.03;
156  }
157  }
158  }
159 
160  if(m_frequency != 0) {
161  m_txEvent = Simulator::Schedule(m_frequency, &SpontaneousProducer::SendTimeout, this);
162  }
163 }

165 void
166 SpontaneousProducer::SendData(const Name &dataName)
167 {
168  if (!m_active)
169  return;
170 
171  //send only ACK (payload size = 1) if payloaded interest was received
172  if (m_subscription == 0 && m_receivedpayload > 0) {
173  m_virtualPayloadSize = 1;
174  }
175 
176  //std::cout << " ack payload= " << m_virtualPayloadSize << std::endl;
177 
178  auto data = make_shared<Data>();
179  data->setName(dataName);
180  data->setFreshnessPeriod(::ndn::time::milliseconds(m_freshness.GetMilliSeconds()));
181  data->setContent(make_shared< ::ndn::Buffer>(m_virtualPayloadSize));
182 
183  Signature signature;
184  SignatureInfo signatureInfo(static_cast< ::ndn::tlv::SignatureTypeValue>(255));
185 
186  if (m_keyLocator.size() > 0) {
187  signatureInfo.setKeyLocator(m_keyLocator);
188  }
189 
190  signature.setInfo(signatureInfo);
191  signature.setValue(::ndn::makeNonNegativeIntegerBlock(::ndn::tlv::SignatureValue, m_signature));
192 
193  data->setSignature(signature);
194 
195  if (m_subscription == 0 && m_receivedpayload > 0) {
196  NS_LOG_INFO("node(" << GetNode()->GetId() << ") sending ACK: " << data->getName() << " TIME: " << Simulator::Now());
197  } else {
198  NS_LOG_INFO("node(" << GetNode()->GetId() << ") sending DATA for " << data->getName() << " TIME: " << Simulator::Now());
199  }
200 
201  //std::cout << "use count " << data.use_count() << " data content/payload size = " << data->getContent().value_size() << std::endl;
202 
203  // to create real wire encoding
204  data->wireEncode();
205 
206  m_transmittedDatas(data, this, m_face);
207  m_appLink->onReceiveData(*data);
208 
209  // Callback for tranmitted subscription data
210  m_sentData(GetNode()->GetId(), data);
211 
212  //std::cout << "APP use count " << data.use_count () << std::endl;
213  //m_appLink->DanFree();
214 }
