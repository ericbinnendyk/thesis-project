161 void
162 ConsumerSync::CheckRetxTimeout()
163 {
164 
165  Time now = Simulator::Now();
166  Time rto = m_rtt->RetransmitTimeout();
167 
168  // NS_LOG_DEBUG ( "Current RTO: " << rto.ToDouble ( Time::S ) << "s" );
169 
170  while ( !m_seqTimeouts.empty() ) {
171 
172  SeqTimeoutsContainer::index<i_timestamp>::type::iterator entry =
173  m_seqTimeouts.get<i_timestamp>().begin();
174 
175  // timeout expired?
176  if ( entry->time + rto <= now ) {
177 
178  uint32_t seqNo = entry->seq;
179  m_seqTimeouts.get<i_timestamp>().erase( entry );
180  OnTimeout( seqNo );
181 
182  } else {
183 
184  break; // nothing else to do. All later packets need not be retransmitted
185  }
186  }
187 
188  m_retxEvent = Simulator::Schedule( m_retxTimer, &ConsumerSync::CheckRetxTimeout, this );
189 }

327 void
328 ConsumerSync::OnData( shared_ptr<const Data> data )
329 {
330 
331  if ( !m_active ) {
332  return;
333  }
334 
335  App::OnData( data ); // tracing inside
336 
337  NS_LOG_FUNCTION( this << data );
338 
339  // This could be a problem......
340  //uint32_t seq = data->getName().at( -1 ).toSequenceNumber();
341 
342  NS_LOG_INFO( "node( " << GetNode()->GetId() << " ) < Received DATA for " << data->getName() << " TIME: " << Simulator::Now() );
343 
344  // Callback for received subscription data
345  m_receivedData( GetNode()->GetId(), data );
346 
347  int hopCount = 0;
348  auto hopCountTag = data->getTag<lp::HopCountTag>();
349 
350  if ( hopCountTag != nullptr ) { // e.g., packet came from local node's cache
351  hopCount = *hopCountTag;
352  }
353 
354  NS_LOG_DEBUG( "Hop count: " << hopCount );
355 
356  //Enable trace file for Interests with sequence number ( subscription = 0 )
357  if ( m_subscription == 0 ) {
358 
359  // This could be a problem......
360  uint32_t seq = data->getName().at( -1 ).toSequenceNumber();
361 
362  SeqTimeoutsContainer::iterator entry = m_seqLastDelay.find( seq );
363 
364  if ( entry != m_seqLastDelay.end() ) {
365  m_lastRetransmittedInterestDataDelay( this, seq, Simulator::Now() - entry->time, hopCount );
366  }
367 
368  entry = m_seqFullDelay.find( seq );
369 
370  if ( entry != m_seqFullDelay.end() ) {
371  m_firstInterestDataDelay( this, seq, Simulator::Now() - entry->time, m_seqRetxCounts[seq], hopCount );
372  }
373 
374  m_seqRetxCounts.erase( seq );
375  m_seqFullDelay.erase( seq );
376  m_seqLastDelay.erase( seq );
377 
378  m_seqTimeouts.erase( seq );
379  m_retxSeqs.erase( seq );
380 
381  m_rtt->AckSeq( SequenceNumber32( seq ) );
382  }
